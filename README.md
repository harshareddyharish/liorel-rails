# README

# Clone the project
1) git clone git@github.com:harshareddyharish/liorel-rails.git

# Goto Project repo
2) cd liorel-rails

# Install the gems
3) bundle install

# Run the migrations 
4) rails db:migrate (or) rake db:migrate

# Run the seed file to create the default user account
5) rails db:seed (or) rake db:seed

# Start the server
6) rails s
